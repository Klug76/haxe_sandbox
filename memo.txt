{
		trace("onContextCreated");

		UiData.WIDTH = stage.stageWidth;
		UiData.HEIGHT = stage.stageHeight;

		_stage3DProxy.context3D.configureBackBuffer(UiData.WIDTH, UiData.HEIGHT, _antiAliasing);

		initAway3D();
		initStarling();
		initListeners();
}
